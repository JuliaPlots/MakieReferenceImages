Trimming grids

using MakieLayout

using AbstractPlotting

scene, layout = layoutscene(resolution = (600, 600))

record(scene, "output.mp4", framerate=1) do io

    ax1 = layout[1, 1] = LAxis(scene, title = "Axis 1")
    recordframe!(io)

    ax2 = layout[1, 2] = LAxis(scene, title = "Axis 2")
    recordframe!(io)

    layout[2, 1] = ax2
    recordframe!(io)

    trim!(layout)
    recordframe!(io)

    layout[2, 3:4] = ax1
    recordframe!(io)

    trim!(layout)
    recordframe!(io)
end
