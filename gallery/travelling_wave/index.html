Travelling wave

using Makie

 scene = Scene()
 mytime = Node(0.0)
 f(v, t) = sin(v + t)
 scene = lines!(
     scene,
     lift(t -> f.(range(0, stop = 2pi, length = 50), t), mytime),
     color = :blue)
 p1 = scene[end];
 N = 100
 record(scene, "output.mp4", range(0, stop = 4pi, length = N)) do i
     mytime[] = i
end
