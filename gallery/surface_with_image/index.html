Surface with image

using Makie

 N = 30
 function xy_data(x, y)
     r = sqrt(x^2 + y^2)
     r == 0.0 ? 1f0 : (sin(r)/r)
 end
 r = range(-2, stop = 2, length = N)
 surf_func(i) = [Float32(xy_data(x*i, y*i)) for x = r, y = r]
 surface(
     r, r, surf_func(10),
     color = rand(RGBAf0, 124, 124)
 )
