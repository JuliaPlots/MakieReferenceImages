Color Legend



 s = surface(0..1, 0..1, rand(100, 100))
 ls = colorlegend(s[end], raw = true, camera = campixel!)
 st = Stepper(vbox(s, ls), "output")
 l = ls[end]
 l[:textcolor] = :blue
 step!(st)
 l[:strokecolor] = :green
 step!(st)
 l[:strokewidth] = 4
 step!(st)
 l[:textsize] = 12
 step!(st)
 l[:textgap] = 5
 step!(st)
 st
